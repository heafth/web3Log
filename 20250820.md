package main

import (
	"fmt"
	"os"
)
//Sprintf 先 "生成字符串" 再由你决定如何使用（打印、存储、传递等）。
func main() {
	t := new(T)
	t.name = "刘德华"
	t.value = 100
	fmt.Println(os.Args)
	fmt.Println(t.name, t.value)
	fmt.Println(t.name + "    " + fmt.Sprint(t.value))
	fmt.Println("Hello, Go!")
}

type T struct {
	name  string // name of the object
	value int    // its value
}
